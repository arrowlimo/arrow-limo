"""
Employee SIN Number Verification Report
Shows which employees are missing SIN numbers for T4 reporting
"""

print("=" * 80)
print("EMPLOYEE SIN NUMBER VERIFICATION REPORT")
print("=" * 80)
print()
print("Run this query in psql or pgAdmin:")
print()
print("-" * 80)

SQL = """
-- Employees missing or invalid SIN numbers
SELECT 
    employee_id,
    employee_name,
    COALESCE(first_name, '(not separate)') as first_name,
    COALESCE(last_name, '(not separate)') as last_name,
    COALESCE(sin, '(MISSING)') as sin_status,
    CASE 
        WHEN sin IS NULL OR sin = '' THEN 'MISSING SIN'
        WHEN LENGTH(REPLACE(REPLACE(sin, ' ', ''), '-', '')) < 9 THEN 'INVALID SIN (too short)'
        ELSE 'OK'
    END as validation,
    employment_status
FROM employees
WHERE sin IS NULL 
   OR sin = '' 
   OR LENGTH(REPLACE(REPLACE(sin, ' ', ''), '-', '')) < 9
ORDER BY employment_status DESC, employee_name;

-- Summary counts
SELECT 
    'TOTAL EMPLOYEES' as category,
    COUNT(*) as count
FROM employees
UNION ALL
SELECT 
    'WITH VALID SIN',
    COUNT(*)
FROM employees
WHERE sin IS NOT NULL 
  AND sin != ''
  AND LENGTH(REPLACE(REPLACE(sin, ' ', ''), '-', '')) >= 9
UNION ALL
SELECT 
    'MISSING SIN',
    COUNT(*)
FROM employees
WHERE sin IS NULL 
   OR sin = ''
UNION ALL  
SELECT
    'INVALID SIN FORMAT',
    COUNT(*)
FROM employees
WHERE sin IS NOT NULL
  AND sin != ''
  AND LENGTH(REPLACE(REPLACE(sin, ' ', ''), '-', '')) < 9
UNION ALL
SELECT 
    'ACTIVE EMPLOYEES',
    COUNT(*)
FROM employees
WHERE employment_status = 'active'
UNION ALL
SELECT
    'ACTIVE MISSING SIN',
    COUNT(*)
FROM employees
WHERE employment_status = 'active'
  AND (sin IS NULL OR sin = '' OR LENGTH(REPLACE(REPLACE(sin, ' ', ''), '-', '')) < 9);
"""

print(SQL)
